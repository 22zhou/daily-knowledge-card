### 基础图像编辑引擎实现

```kotlin
/**
 * Android图像编辑核心引擎
 * 负责管理图像状态、变换操作和渲染
 */
class ImageEditorEngine(private val context: Context) {
    
    // 当前编辑的位图（使用不可变副本保证数据安全）
    private var originalBitmap: Bitmap? = null
    private var workingBitmap: Bitmap? = null
    
    // 变换矩阵用于缩放、旋转、平移
    private val transformMatrix = Matrix()
    
    // 编辑操作历史（支持撤销/重做）
    private val undoStack = mutableListOf<EditOperation>()
    private val redoStack = mutableListOf<EditOperation>()
    
    // 裁剪区域
    private var cropRect = RectF()
    
    /**
     * 加载图片
     * @param uri 图片URI
     * @param maxSize 最大边长限制，防止OOM
     */
    fun loadImage(uri: Uri, maxSize: Int = 2048): Boolean {
        return try {
            // 使用BitmapFactory.Options进行内存友好的图片加载
            val options = BitmapFactory.Options().apply {
                // 第一次解码只获取尺寸信息
                inJustDecodeBounds = true
            }
            
            context.contentResolver.openInputStream(uri)?.use { input ->
                BitmapFactory.decodeStream(input, null, options)
            }
            
            // 计算采样率
            options.inSampleSize = calculateInSampleSize(
                options.outWidth, 
                options.outHeight, 
                maxSize, 
                maxSize
            )
            options.inJustDecodeBounds = false
            options.inPreferredConfig = Bitmap.Config.ARGB_8888
            
            // 第二次解码获取实际位图
            context.contentResolver.openInputStream(uri)?.use { input ->
                val decoded = BitmapFactory.decodeStream(input, null, options)
                if (decoded != null) {
                    originalBitmap = decoded
                    workingBitmap = decoded.copy(Bitmap.Config.ARGB_8888, true)
                    resetTransform()
                    true
                } else false
            } ?: false
        } catch (e: Exception) {
            Log.e("ImageEditor", "Failed to load image", e)
            false
        }
    }
    
    /**
     * 计算合适的采样率
     * 在保证显示质量的同时最小化内存占用
     */
    private fun calculateInSampleSize(
        width: Int, height: Int, reqWidth: Int, reqHeight: Int
    ): Int {
        var inSampleSize = 1
        if (height > reqHeight || width > reqWidth) {
            val halfHeight = height / 2
            val halfWidth = width / 2
            // 取2的幂次方，便于GPU处理
            while ((halfHeight / inSampleSize) >= reqHeight 
                   && (halfWidth / inSampleSize) >= reqWidth) {
                inSampleSize *= 2
            }
        }
        return inSampleSize
    }
    
    /**
     * 应用滤镜效果
     * @param filterType 滤镜类型
     */
    fun applyFilter(filterType: FilterType) {
        val bitmap = workingBitmap ?: return
        
        // 保存当前状态用于撤销
        val previousState = bitmap.copy(Bitmap.Config.ARGB_8888, false)
        
        // 创建画布进行滤镜绘制
        val filteredBitmap = bitmap.copy(Bitmap.Config.ARGB_8888, true)
        val canvas = Canvas(filteredBitmap)
        
        // 根据滤镜类型应用不同效果
        val paint = Paint().apply {
            colorFilter = when (filterType) {
                FilterType.GRAYSCALE -> ColorMatrixColorFilter(
                    ColorMatrix().apply { setSaturation(0f) }
                )
                FilterType.SEPIA -> createSepiaMatrix()
                FilterType.WARM -> createWarmMatrix()
                FilterType.COOL -> createCoolMatrix()
                FilterType.VIBRANT -> createVibrantMatrix()
                FilterType.NONE -> null
            }
        }
        
        canvas.drawBitmap(bitmap, 0f, 0f, paint)
        
        // 更新工作位图并记录操作
        workingBitmap = filteredBitmap
        undoStack.add(FilterOperation(filterType, previousState))
        redoStack.clear()
    }
    
    /**
     * 创建暖色调滤镜矩阵
     * 增加红色和黄色，减少蓝色
     */
    private fun createWarmMatrix(): ColorMatrixColorFilter {
        val colorMatrix = ColorMatrix().apply {
            // 提高红色饱和度
            val warmMatrix = floatArrayOf(
                1.2f, 0f, 0f, 0f, 20f,   // Red
                0f, 1.1f, 0f, 0f, 10f,   // Green  
                0f, 0f, 0.9f, 0f, -20f, // Blue
                0f, 0f, 0f, 1f, 0f       // Alpha
            )
            set(warmMatrix)
        }
        return ColorMatrixColorFilter(colorMatrix)
    }
    
    /**
     * 创建冷色调滤镜矩阵
     */
    private fun createCoolMatrix(): ColorMatrixColorFilter {
        val colorMatrix = ColorMatrix().apply {
            val coolMatrix = floatArrayOf(
                0.9f, 0f, 0f, 0f, -20f,  // Red
                0f, 1.0f, 0f, 0f, 0f,    // Green
                0f, 0f, 1.2f, 0f, 20f,   // Blue
                0f, 0f, 0f, 1f, 0f       // Alpha
            )
            set(coolMatrix)
        }
        return ColorMatrixColorFilter(colorMatrix)
    }
    
    /**
     * 裁剪图片
     * @param rect 裁剪区域（相对于原图）
     */
    fun cropImage(rect: RectF): Boolean {
        val bitmap = workingBitmap ?: return false
        
        val intRect = Rect(
            rect.left.toInt().coerceIn(0, bitmap.width),
            rect.top.toInt().coerceIn(0, bitmap.height),
            rect.right.toInt().coerceIn(0, bitmap.width),
            rect.bottom.toInt().coerceIn(0, bitmap.height)
        )
        
        if (intRect.width() < 10 || intRect.height() < 10) {
            return false
        }
        
        // 保存当前状态
        val previousState = bitmap.copy(Bitmap.Config.ARGB_8888, false)
        
        // 执行裁剪
        val croppedBitmap = Bitmap.createBitmap(
            bitmap,
            intRect.left,
            intRect.top,
            intRect.width(),
            intRect.height()
        )
        
        workingBitmap = croppedBitmap
        previousState.recycle() // 释放旧位图内存
        
        // 重置变换矩阵
        resetTransform()
        
        undoStack.add(CropOperation(rect, previousState))
        redoStack.clear()
        
        return true
    }
    
    /**
     * 旋转图片
     * @param degrees 旋转角度（90的倍数）
     */
    fun rotate(degrees: Float) {
        val bitmap = workingBitmap ?: return
        val previousState = bitmap.copy(Bitmap.Config.ARGB_8888, false)
        
        val matrix = Matrix().apply {
            postRotate(degrees, bitmap.width / 2f, bitmap.height / 2f)
        }
        
        val rotatedBitmap = Bitmap.createBitmap(
            bitmap, 0, 0, bitmap.width, bitmap.height, matrix, true
        )
        
        workingBitmap = rotatedBitmap
        previousState.recycle()
        
        undoStack.add(RotateOperation(degrees, previousState))
        redoStack.clear()
    }
    
    /**
     * 撤销操作
     */
    fun undo(): Boolean {
        if (undoStack.isEmpty()) return false
        
        val operation = undoStack.removeLast()
        redoStack.add(operation)
        
        // 恢复之前状态
        workingBitmap?.recycle()
        workingBitmap = operation.previousState.copy(Bitmap.Config.ARGB_8888, true)
        
        return true
    }
    
    /**
     * 重做操作
     */
    fun redo(): Boolean {
        if (redoStack.isEmpty()) return false
        
        val operation = redoStack.removeLast()
        undoStack.add(operation)
        
        workingBitmap?.recycle()
        workingBitmap = operation.previousState.copy(Bitmap.Config.ARGB_8888, true)
        
        return true
    }
    
    /**
     * 导出编辑后的图片
     * @param quality 导出质量（0-100）
     * @param format 图片格式
     */
    fun export(quality: Int = 95, format: ExportFormat = ExportFormat.JPEG): ByteArray? {
        val bitmap = workingBitmap ?: return null
        
        val compressFormat = when (format) {
            ExportFormat.JPEG -> Bitmap.CompressFormat.JPEG
            ExportFormat.PNG -> Bitmap.CompressFormat.PNG
            ExportFormat.WEBP -> Bitmap.CompressFormat.WEBP_LOSSY
        }
        
        val outputStream = ByteArrayOutputStream()
        bitmap.compress(compressFormat, quality, outputStream)
        
        return outputStream.toByteArray()
    }
    
    /**
     * 重置变换矩阵
     */
    private fun resetTransform() {
        transformMatrix.reset()
        originalBitmap?.let {
            // 将变换重置为适应屏幕大小
            transformMatrix.postScale(
                1f, 1f
            )
        }
    }
    
    /**
     * 释放资源
     */
    fun release() {
        workingBitmap?.recycle()
        workingBitmap = null
        originalBitmap?.recycle()
        originalBitmap = null
        undoStack.clear()
        redoStack.clear()
    }
}

// 滤镜类型枚举
enum class FilterType {
    NONE, GRAYSCALE, SEPIA, WARM, COOL, VIBRANT
}

// 导出格式枚举
enum class ExportFormat {
    JPEG, PNG, WEBP
}

// 编辑操作基类
sealed class EditOperation {
    abstract val previousState: Bitmap
}

// 滤镜操作
class FilterOperation(
    val filterType: FilterType,
    override val previousState: Bitmap
) : EditOperation()

// 裁剪操作
class CropOperation(
    val cropRect: RectF,
    override val previousState: Bitmap
) : EditOperation()

// 旋转操作
class RotateOperation(
    val degrees: Float,
    override val previousState: Bitmap
) : EditOperation()
```

## 4.2 手势缩放与平移实现

```kotlin
/**
 * 自定义ImageView，支持手势缩放和平移
 * 基于ScaleGestureDetector和GestureDetector实现
 */
class ScalableImageView @JvmOverloads constructor(
    context: Context,
    attrs: AttributeSet? = null,
    defStyleAttr: Int = 0
) : AppCompatImageView(context, attrs, defStyleAttr) {

    // 缩放手势检测器
    private val scaleGestureDetector: ScaleGestureDetector
    
    // 双击/滑动手势检测器
    private val gestureDetector: GestureDetector
    
    // 当前变换矩阵
    private val imageMatrix = Matrix()
    
    // 变换矩阵的值数组（用于获取具体数值）
    private val matrixValues = FloatArray(9)
    
    // 当前缩放比例
    private var currentScale = 1f
    
    // 最小/最大缩放比例
    private var minScale = 1f
    private var maxScale = 5f
    
    // 手指数量
    private var touchCount = 0
    
    // 上一次变换矩阵
    private val savedMatrix = Matrix()
    
    // 模式：NONE-无操作, DRAG-拖拽, ZOOM-缩放
    private var mode = Mode.NONE
    
    // 初始两点之间的距离
    private var oldDist = 1f
    
    // 初始焦点
    private var focusX = 0f
    private var focusY = 0f

    enum class Mode {
        NONE, DRAG, ZOOM
    }
    
    init {
        // 设置缩放类型为矩阵
        scaleType = ScaleType.MATRIX
        
        // 初始化手势检测器
        scaleGestureDetector = ScaleGestureDetector(
            context,
            object : ScaleGestureDetector.SimpleOnScaleGestureListener() {
                override fun onScale(detector: ScaleGestureDetector): Boolean {
                    var scaleFactor = detector.scaleFactor
                    
                    // 计算新的缩放比例
                    val newScale = currentScale * scaleFactor
                    
                    // 限制缩放范围
                    if (newScale in minScale..maxScale) {
                        currentScale = newScale
                        
                        // 以焦点为中心缩放
                        imageMatrix.postScale(
                            scaleFactor, scaleFactor,
                            detector.focusX, detector.focusY
                        )
                        
                        setImageMatrix(imageMatrix)
                    }
                    
                    return true
                }
            }
        )
        
        gestureDetector = GestureDetector(
            context,
            object : GestureDetector.SimpleOnGestureListener() {
                override fun onDoubleTap(e: MotionEvent): Boolean {
                    // 双击切换缩放状态
                    if (currentScale > minScale + 0.1f) {
                        // 缩小回原始大小
                        animate()
                            .scaleX(1f)
                            .scaleY(1f)
                            .translationX(0f)
                            .translationY(0f)
                            .setDuration(300)
                            .start()
                        currentScale = minScale
                    } else {
                        // 放大到最大
                        animate()
                            .scaleX(2.5f)
                            .scaleY(2.5f)
                            .setDuration(300)
                            .start()
                        currentScale = 2.5f
                    }
                    return true
                }
            }
        )
    }
    
    override fun onTouchEvent(event: MotionEvent): Boolean {
        // 委托给缩放手势检测器
        scaleGestureDetector.onTouchEvent(event)
        
        // 委托给手势检测器（处理双击等）
        gestureDetector.onTouchEvent(event)
        
        when (event.action and MotionEvent.ACTION_MASK) {
            MotionEvent.ACTION_DOWN -> {
                touchCount = 1
                savedMatrix.set(imageMatrix)
                focusX = event.x
                focusY = event.y
                mode = Mode.DRAG
            }
            
            MotionEvent.ACTION_POINTER_DOWN -> {
                touchCount = 2
                oldDist = spacing(event)
                if (oldDist > 10f) {
                    savedMatrix.set(imageMatrix)
                    midPoint(focusX, focusY, event)
                    mode = Mode.ZOOM
                }
            }
            
            MotionEvent.ACTION_UP, MotionEvent.ACTION_POINTER_UP -> {
                touchCount = 0
                mode = Mode.NONE
            }
            
            MotionEvent.ACTION_MOVE -> {
                if (mode == Mode.DRAG && touchCount == 1) {
                    // 单指拖拽
                    imageMatrix.set(savedMatrix)
                    val dx = event.x - focusX
                    val dy = event.y - focusY
                    imageMatrix.postTranslate(dx, dy)
                    setImageMatrix(imageMatrix)
                }
            }
        }
        
        // 边界检测，防止图片移出屏幕
        constrainMatrix()
        
        return true
    }
    
    /**
     * 计算两点之间的距离
     */
    private fun spacing(event: MotionEvent): Float {
        if (event.pointerCount < 2) return 0f
        val x = event.getX(0) - event.getX(1)
        val y = event.getY(0) - event.getY(1)
        return sqrt(x * x + y * y)
    }
    
    /**
     * 计算两点的中点
     */
    private fun midPoint(point: FloatArray, event: MotionEvent) {
        if (event.pointerCount < 2) return
        point[0] = (event.getX(0) + event.getX(1)) / 2
        point[1] = (event.getY(0) + event.getY(1)) / 2
    }
    
    /**
     * 限制矩阵，确保图片不超出边界
     */
    private fun constrainMatrix() {
        imageMatrix.getValues(matrixValues)
        val transX = matrixValues[Matrix.MTRANS_X]
        val transY = matrixValues[Matrix.MTRANS_Y]
        val scale = matrixValues[Matrix.MSCALE_X]
        
        val drawable = drawable ?: return
        
        val drawableWidth = drawable.intrinsicWidth * scale
        val drawableHeight = drawable.intrinsicHeight * scale
        
        // 计算最大平移量
        val maxTransX = if (drawableWidth < width) {
            (width - drawableWidth) / 2
        } else {
            0f
        }
        val maxTransY = if (drawableHeight < height) {
            (height - drawableHeight) / 2
        } else {
            0f
        }
        
        // 应用边界限制
        val newTransX = transX.coerceIn(
            width - drawableWidth - maxTransX,
            maxTransX
        )
        val newTransY = transY.coerceIn(
            height - drawableHeight - maxTransY,
            maxTransY
        )
        
        if (transX != newTransX || transY != newTransY) {
            imageMatrix.postTranslate(newTransX - transX, newTransY - transY)
            setImageMatrix(imageMatrix)
        }
    }

     * 设置图片并适应屏幕
     */
    fun setImageFitCenter(bitmap: Bitmap) {
        // 计算居中缩放
        val viewWidth = width.toFloat()
        val viewHeight = height.toFloat()
        
        if (viewWidth == 0f || viewHeight == 0f) {
            post { setImageFitCenter(bitmap) }
            return
        }
        
        val bitmapWidth = bitmap.width.toFloat()
        val bitmapHeight = bitmap.height.toFloat()
        
        // 计算缩放比例使图片适应View
        val scaleX = viewWidth / bitmapWidth
        val scaleY = viewHeight / bitmapHeight
        val scale = minOf(scaleX, scaleY)
        
        // 计算平移量使图片居中
        val dx = (viewWidth - bitmapWidth * scale) / 2
        val dy = (viewHeight - bitmapHeight * scale) / 2
        
        imageMatrix.reset()
        imageMatrix.setScale(scale, scale)
        imageMatrix.postTranslate(dx, dy)
        
        setImageMatrix(imageMatrix)
        currentScale = scale
        minScale = scale
    }
```

### 1. **算法：接雨水问题**

```python
"""
LeetCode 42. 接雨水
给定 n 个非负整数表示每个宽度为 1 的柱子的高度图，
计算按此排列的柱子，下雨之后能接多少雨水。
"""
def trap(height: list[int]) -> int:
    """
    双指针解法
    时间复杂度: O(n)
    空间复杂度: O(1)
    """
    if not height:
        return 0
    
    left, right = 0, len(height) - 1
    left_max, right_max = 0, 0
    water = 0
    
    while left < right:
        if height[left] < height[right]:
            # 左边柱子更矮，取决于左边最大高度
            if height[left] >= left_max:
                left_max = height[left]
            else:
                water += left_max - height[left]
            left += 1
        else:
            # 右边柱子更矮，取决于右边最大高度
            if height[right] >= right_max:
                right_max = height[right]
            else:
                water += right_max - height[right]
            right -= 1
    
    return water

# 测试
print(trap([0,1,0,2,1,0,1,3,2,1,2,1]))  # 输出: 6
```
